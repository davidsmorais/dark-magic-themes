import { themeRegistry } from './theme-registry';
type ThemeSummary = {
name: string;
accents: string[];
};

export function renderPreview(themeName: string) {
const activeTheme = themeRegistry.find(
(theme) => theme.label === themeName
);
// Keep the UI expressive without adding extra chrome.
return `Previewing ${activeTheme?.label ?? 'Dark Magic'}`;
}
